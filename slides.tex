%--- Slide 1: Motivation (China hook) -------------------------------------
\begin{frame}{China, 1966--1989: A Cultural Arc}

  \textbf{The Cultural Revolution (1966--76): erasure of private life}
  \begin{itemize}
    \item Millions killed, imprisoned or sent to labor camps (MacFarquhar \& Schoenhals 2006)
    \item 16 million urban youth forcibly relocated to the countryside (Bernstein 1977; Xu 2013)
    \item Private enterprise, religious practice, and family-based farming abolished
    %\item the individual's economic fate entirely determined by collective assignment
  \end{itemize}

  \vspace{0.6em}
  \textbf{1978 Reforms: the fastest de-collectivisation in history}
  \begin{itemize}
    \item By 1984, \textbf{99\%} of farm households transitioned to individual farming under the Household Responsibility System (Lin, \textit{AER} 1992)
    \item Private enterprises: from essentially zero in 1978 to \textbf{22 million} registered by 1993 (Naughton 2007)
  \end{itemize}

  \vspace{0.6em}
  \textit{How did cultural values respond to the sharp transformations?}
\end{frame}

%--- Slide 2: This Paper --------------------------------------------------
\begin{frame}{This Paper}
  \textbf{This project:} use \alert{song lyrics} as a real-time, mass-produced cultural barometer
  \begin{itemize}
    \item Low cost of circulation; no literacy required
    \item Sentiment-laden by design
    \item Informative precisely where surveys are not
    \item A conduit for ideas even when other channels were thin
    \item Especially important in the Chinese context
  \end{itemize}
  \vspace{3pt}
\begin{block}{Research Question}
  \begin{itemize}
    \item Document temporal trend of individualism from 1950-2000 
    \item Did China's post-1978 economic reforms \textbf{increase individualism} in popular culture, captured in pop song lyrics?
  \end{itemize}
  
\end{block}
\end{frame}

    \begin{itemize}
    \item A cultural trait attributed to developed societies but considered absent in pre-modern ones

%--- Slide 3a: Pre-Reform Context ------------------------------------------
\begin{frame}{Pre-Reform China: The Command Economy}

\begin{itemize}\setlength\itemsep{0.7em}
  \item \textbf{Danwei (work-unit) system:} employment, housing, healthcare, and social
    identity all assigned by the state — the individual's life determined entirely by
    collective membership
  \item \textbf{People's communes:} collective farming abolished private land; output
    distributed by work-points, not individual effort (Lin, \textit{AER} 1992)
  \item \textbf{Cultural Revolution (1966–76):} private enterprise and family-based loyalty
    subordinated to party collective; 16 million urban youth forcibly relocated to the
    countryside (Bernstein 1977)
  \item \textbf{Cultural production:} all music, literature, and film produced and
    distributed through state apparatus — individual expression subordinated to
    collective ideology
\end{itemize}

\end{frame}

%--- Slide 3b: Post-Reform Context -----------------------------------------
\begin{frame}{Post-Reform China: Market Liberalisation}

\begin{itemize}\setlength\itemsep{0.7em}
  \item \textbf{Household Responsibility System:} by 1984, 99\% of farm households
    contracted individual plots — restoring personal economic agency after decades of
    collective farming (Lin, \textit{AER} 1992)
  \item \textbf{Labour mobility:} hukou relaxation enabled large-scale internal migration;
    workers made individual choices about where and how to work
  \item \textbf{Private enterprise:} from essentially zero to 22 million registered firms
    by 1993 — individual risk-taking rewarded for the first time (Naughton 2007)
  \item \textbf{``Iron rice bowl'' fractured:} state guarantees of lifetime employment
    withdrawn — individual initiative, risk, and reward entered everyday life
\end{itemize}

\vspace{0.5em}
\begin{center}
  \textit{Restructured economic incentives $\Rightarrow$ reshaped preferences $\Rightarrow$ cultural change}
\end{center}
\end{frame}

%--- Slide 6: Measurement Overview -----------------------------------------
\begin{frame}[label=slide-dictmeasures]{Measuring Individualism: Dictionary-based Approaches}

Goal: continuous score per song $\in [-1,+1]$ that consistently captures individualism.

\vspace{0.4em}
  \begin{itemize}
    \item \textbf{Raw Count Net Score}
      $$S_{\text{raw}} = \frac{I - C}{I + C}$$
      Counts of dictionary hits for individualism ($I$) and collectivism ($C$).
    \vspace{0.3em}
    \item \textbf{TF-IDF Net Score}
      $$S_{\text{tfidf}} = \frac{W_I - W_C}{W_I + W_C}$$
      TF-IDF weights over dictionary terms. Down-weights ubiquitous words (e.g., ``we'').
  \end{itemize}
\end{frame}

\begin{frame}{Normalization}
  \textbf{Why this normalisation?}
    \footnotesize
    \begin{enumerate}
      \item \textbf{Controls for song length:} dividing by $I+C$ measures relative balance, not raw counts
      \item \textbf{Bounded in $[-1,+1]$:} pure individualist $= +1$; pure collectivist $= -1$; no dictionary hits $= 0$
      \item \textbf{Symmetric:} same denominator on both sides — no directional inflation
    \end{enumerate}

%--- Slide 10: Interpreting the Break -------------------------------------
\begin{frame}{What Drives the Break in Trend?}

\begin{enumerate}\setlength\itemsep{0.8em}
  \item \textbf{Change in cultural production pattern}
    \begin{itemize}
      \item \textit{From centralised to decentralised production:} pre-1978, all recording
        and distribution controlled by the state-owned China Record Company; post-1978,
        regional labels emerged (esp.\ Guangdong), cassette tape distribution bypassed
        state channels, and Hong Kong/Taiwan music entered the mainland market
      \item \textit{Loosening of censorship:} during the Cultural Revolution (1966--76)
        only 8 approved \textit{yangbanxi} (model operas) were permitted and all personal
        or romantic content was banned; post-1978, romantic and personal themes were
        gradually allowed, and despite the 1983 Anti-Spiritual Pollution Campaign
        enforcement weakened through the decade
    \end{itemize}

  \item \textbf{Changes in economic institutions spurred individualism}
\end{enumerate}

\end{frame}

%--- Slide 11: Identification Strategy ------------------------------------
\begin{frame}{Empirical Strategy: Identification}

\textbf{Core idea:} exploit \textbf{differential adoption} of market reforms across provinces via the 1982–84 cadre replacement campaign.

\vspace{0.4em}
\textbf{Treatment — \texttt{ReformExposure}$_{p,y}$:}
\begin{itemize}
  \item Percentage decline in provincial leaders from the \textit{Mao cohort} during the 1982–84 turnover window
  \item $\uparrow$ \texttt{ReformExposure}$_{p,y}$ $\Rightarrow$ $\uparrow$ displacement of Mao-era leadership $\Rightarrow$ \alert{$\uparrow$ reform exposure}
  \item Interacted with a post-1982 indicator
\end{itemize}

\vspace{0.4em}
\textbf{Key identifying assumption} \\ $\Rightarrow$
cadre replacement should not be correlated with local cultural development
\vspace{-0.4em}
\begin{itemize}
  \item Top-down, not driven by local conditions
  \item Institutional response to political screening constraints
\end{itemize}

\end{frame}

%--- Slide 10: Specification -----------------------------------------------
\begin{frame}{Empirical Strategy: Specification}

\textbf{Baseline regression:}
\begin{equation*}
  \text{Individualism}_{s(p,y)} = \beta \cdot \texttt{ReformExposure}_{p,y} + \text{Post}_y \cdot \ln\text{GDP}_{1981,p} + \lambda_p + \sigma_y + \varepsilon
\end{equation*}
\vspace{-0.3em}
where $s(p,y)$ indexes song $s$ produced in province $p$ in year $y$.

\vspace{0.8em}
\textbf{Alternative Hypothesis}
\begin{itemize}
  \item Nationwide ideological relaxation post-Cultural Revolution would produce synchronous increases across all provinces — absorbed by \textbf{year fixed effects}.
  \item A comeback of sentiments: more open places express higher individualism unconstrained, post-relaxation \\
    $\hookrightarrow$ but cadre turnover should be exogeneous
\end{itemize}
\end{frame}

%--- Slide 13: Discussion --------------------------------------------------
\begin{frame}{Discussion \& Interpretation}

  \textbf{Summary of findings:}
  \begin{itemize}
    \item Robust positive relationship between reform intensity and lyrical individualism across three of four measures
  \item Negative coeff. on GDP interaction \\
  $\Rightarrow$ Provinces that were better-off during the CR were \emph{also} those with higher collectivism, and experienced a \emph{smaller} individualism rise under reform. Suggests reform operated through structural change rather than income growth alone.
  \end{itemize}
\end{frame}
